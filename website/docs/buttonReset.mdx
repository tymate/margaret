ButtonReset is a button component without style

Example

import { ButtonReset } from '@tymate/margaret';

<div
  style={{
    border: '1px solid #dedede',
    width: '300px',
    padding: '10px',
    marginBottom: '20px',
  }}
>
  <ButtonReset
    size="full"
    style={{ border: '1px solid #dedede', padding: '10px' }}
  >
    Reset
  </ButtonReset>
</div>
